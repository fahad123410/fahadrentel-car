📜 Firebase Security Rules (Copy to Firestore & Storage)
These rules allow public read for cars/banners, public create for bookings, and file uploads.
// FIRESTORE RULES v2
rules_version = '2';
service cloud.firestore {
match /databases/{database}/documents {
match /cars/{doc} { allow read: if true; allow write: if false; }
match /banners/{doc} { allow read: if true; allow write: if false; }
match /settings/{doc} { allow read: if true; allow write: if false; }
match /bookings/{doc} {
allow read: if true;
allow create: if true;
allow update: if request.resource.data.diff(resource.data).affectedKeys().hasOnly(['amount_paid','balance_due','payment_status','payment_proof_url','payment_method','status']);
allow delete: if false;
}
}
}
// STORAGE RULES
service firebase.storage {
match /b/{bucket}/o {
match /id_cards/{file} { allow read: if true; allow write: if request.auth == null; }
match /licenses/{file} { allow read: if true; allow write: if request.auth == null; }
match /payment_proofs/{file} { allow read: if true; allow write: if request.auth == null; }
}
}
